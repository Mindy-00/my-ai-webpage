    <script>

        /**

         * 宇宙星空爆炸系统 - 锁定头部区域

         */

        const canvas = document.getElementById('starfield-canvas');

        const ctx = canvas.getContext('2d');

        const container = document.querySelector('.hero-section');

        

        let particles = [];

        let stars = [];

        let mouse = { x: -100, y: -100 };

        const particleCount = 200;

        

        const shapes = {

            random: () => ({ x: Math.random() * canvas.width, y: Math.random() * canvas.height }),

            circle: (i) => {

                const angle = (i / particleCount) * Math.PI * 2;

                const r = Math.min(canvas.width, canvas.height) * 0.28;

                return { x: canvas.width/2 + Math.cos(angle) * r, y: canvas.height/2 + Math.sin(angle) * r };

            },

            square: (i) => {

                const side = particleCount / 4;

                const r = Math.min(canvas.width, canvas.height) * 0.25;

                let x, y;

                if (i < side) { x = -r + (i/side)*2*r; y = -r; }

                else if (i < side*2) { x = r; y = -r + ((i-side)/side)*2*r; }

                else if (i < side*3) { x = r - ((i-side*2)/side)*2*r; y = r; }

                else { x = -r; y = r - ((i-side*3)/side)*2*r; }

                return { x: canvas.width/2 + x, y: canvas.height/2 + y };

            }

        };


        let currentShapeMode = 'random';


        class Particle {

            constructor(i) {

                this.index = i;

                this.reset();

            }

            reset() {

                this.x = Math.random() * canvas.width;

                this.y = Math.random() * canvas.height;

                this.vx = (Math.random() - 0.5) * 2;

                this.vy = (Math.random() - 0.5) * 2;

                this.accel = 0.04 + Math.random() * 0.08;

                this.friction = 0.92;

                this.size = Math.random() * 2.5 + 0.5;

                this.color = `rgba(255, 255, 255, ${Math.random() * 0.6 + 0.2})`;

            }

            explode(mx, my) {

                const dx = this.x - mx;

                const dy = this.y - my;

                const dist = Math.sqrt(dx*dx + dy*dy);

                if (dist < 500) {

                    const force = (500 - dist) / 500;

                    const angle = Math.atan2(dy, dx);

                    this.vx += Math.cos(angle) * force * 45;

                    this.vy += Math.sin(angle) * force * 45;

                }

            }

            update() {

                const target = shapes[currentShapeMode](this.index);

                this.vx += (target.x - this.x) * this.accel * 0.1;

                this.vy += (target.y - this.y) * this.accel * 0.1;

                this.vx *= this.friction;

                this.vy *= this.friction;

                this.x += this.vx;

                this.y += this.vy;

            }

            draw() {

                ctx.fillStyle = this.color;

                ctx.beginPath();

                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);

                ctx.fill();

                if (this.size > 2) {

                    ctx.shadowBlur = 10;

                    ctx.shadowColor = "white";

                    ctx.fill();

                    ctx.shadowBlur = 0;

                }

            }

        }


        function init() {

            canvas.width = container.offsetWidth;

            canvas.height = container.offsetHeight;

            stars = [];

            for(let i=0; i<150; i++) {

                stars.push({

                    x: Math.random() * canvas.width,

                    y: Math.random() * canvas.height,

                    size: Math.random() * 1.2,

                    opacity: Math.random() * 0.4

                });

            }

            particles = [];

            for (let i = 0; i < particleCount; i++) {

                particles.push(new Particle(i));

            }

        }


        function animate() {

            ctx.fillStyle = 'rgba(0, 0, 0, 0.18)';

            ctx.fillRect(0, 0, canvas.width, canvas.height);

            stars.forEach(s => {

                ctx.fillStyle = `rgba(255, 255, 255, ${s.opacity})`;

                ctx.beginPath();

                ctx.arc(s.x, s.y, s.size, 0, Math.PI * 2);

                ctx.fill();

            });

            particles.forEach(p => { p.update(); p.draw(); });

            ctx.lineWidth = 0.4;

            for (let i = 0; i < particles.length; i++) {

                for (let j = i + 1; j < particles.length; j++) {

                    const dx = particles[i].x - particles[j].x;

                    const dy = particles[i].y - particles[j].y;

                    const dist = Math.sqrt(dx*dx + dy*dy);

                    if (dist < 75) {

                        ctx.strokeStyle = `rgba(255, 255, 255, ${0.12 * (1 - dist/75)})`;

                        ctx.beginPath();

                        ctx.moveTo(particles[i].x, particles[i].y);

                        ctx.lineTo(particles[j].x, particles[j].y);

                        ctx.stroke();

                    }

                }

            }

            requestAnimationFrame(animate);

        }


        container.addEventListener('mousemove', (e) => {

            const rect = canvas.getBoundingClientRect();

            mouse.x = e.clientX - rect.left;

            mouse.y = e.clientY - rect.top;

        });


        container.addEventListener('mousedown', (e) => {

            const rect = canvas.getBoundingClientRect();

            const mx = e.clientX - rect.left;

            const my = e.clientY - rect.top;

            particles.forEach(p => p.explode(mx, my));

            const modes = ['random', 'circle', 'square'];

            const currentIndex = modes.indexOf(currentShapeMode);

            currentShapeMode = modes[(currentIndex + 1) % modes.length];

        });


        window.addEventListener('resize', init);

        init();

        animate();


        // 页面动效处理

        gsap.registerPlugin(ScrollTrigger);


        // 头部进入动画

        const tl = gsap.timeline();

        tl.to("#hero-badge", { opacity: 1, y: 0, duration: 1, delay: 0.5 })

          .to("#hero-title", { opacity: 1, y: 0, duration: 1 }, "-=0.7")

          .to("#hero-desc", { opacity: 1, y: 0, duration: 1 }, "-=0.7")

          .to("#hero-btns", { opacity: 1, y: 0, duration: 1 }, "-=0.7");


        // 滚动揭示动画

        gsap.utils.toArray('.reveal').forEach(elem => {

            gsap.fromTo(elem, 

                { opacity: 0, y: 50 },

                {

                    opacity: 1, y: 0, duration: 1, ease: "power3.out",

                    scrollTrigger: {

                        trigger: elem,

                        start: "top bottom-=100",

                        toggleActions: "play none none none"

                    }

                }

            );

        });


        // 技能条增长动画

        ScrollTrigger.create({

            trigger: "#about",

            start: "top center",

            onEnter: () => {

                document.querySelectorAll('.skill-bar-fill').forEach(bar => {

                    bar.style.width = bar.getAttribute('data-width');

                });

            }

        });
